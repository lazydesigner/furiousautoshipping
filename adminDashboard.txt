Quote {
  id                String   @id @default(cuid())
  trackingId        String   @unique
  createdAt         DateTime @default(now())
  updatedAt         DateTime @updatedAt
  
  // Customer Info
  name              String
  phone             String  
  email             String
  
  // Route Info
  originAddress     String
  destinationAddress String
  originCity        String
  originState       String
  destinationCity   String
  destinationState  String
  distance          Float?
  
  // Pricing
  baseCost          Float?
  totalCost         Float?
  costBreakdown     Json?
  
  // Status & Assignment
  status            QuoteStatus @default(NEW)
  assignedTo        String?     // Admin user ID
  carrierId         String?     // Assigned carrier
  followUpDate      DateTime?
  priority          Priority    @default(NORMAL)
  
  // Dates
  pickupDate        DateTime?
  deliveryDate      DateTime?
  estimatedDelivery DateTime?
  
  // Service Details
  transportType     TransportType
  pickupType        PickupType
  expedited         Boolean @default(false)
  seasonalService   Boolean @default(false)
  
  // Business Metrics
  leadSource        String?
  conversionDate    DateTime?
  revenue           Float?
  profit            Float?
  
  // Relations
  vehicles          Vehicle[]
  notes             QuoteNote[]
  documents         Document[]
  payments          Payment[]
  
  @@map("quotes")
}

model QuoteNote {
  id        String   @id @default(cuid())
  quoteId   String
  quote     Quote    @relation(fields: [quoteId], references: [id])
  userId    String   // Admin who added note
  content   String
  isInternal Boolean @default(true)
  createdAt DateTime @default(now())
  
  @@map("quote_notes")
}

model Carrier {
  id            String   @id @default(cuid())
  name          String
  contactPerson String
  phone         String
  email         String
  dotNumber     String   @unique
  mcNumber      String?
  insuranceExp  DateTime
  rating        Float?
  activeRoutes  String[] // JSON array of state codes
  capacity      Int
  available     Boolean  @default(true)
  
  quotes        Quote[]
  
  @@map("carriers")
}

model Payment {
  id          String        @id @default(cuid())
  quoteId     String
  quote       Quote         @relation(fields: [quoteId], references: [id])
  amount      Float
  type        PaymentType   // DEPOSIT, BALANCE, FULL
  status      PaymentStatus @default(PENDING)
  method      String?       // Cash, Check, Credit
  paidAt      DateTime?
  dueDate     DateTime?
  
  @@map("payments")
}

// Enums
enum QuoteStatus {
  NEW
  CONTACTED  
  QUOTED
  NEGOTIATING
  ACCEPTED
  CARRIER_ASSIGNED
  PICKED_UP
  IN_TRANSIT
  DELIVERED
  COMPLETED
  CANCELLED
  FOLLOW_UP
}

enum Priority {
  LOW
  NORMAL  
  HIGH
  URGENT
}

enum PaymentType {
  DEPOSIT
  BALANCE
  FULL
}

enum PaymentStatus {
  PENDING
  PAID
  OVERDUE
  CANCELLED
}
